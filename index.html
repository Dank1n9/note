int arr[3]={10,20,30}; cout<<arr; //输出第一个元素的内存地址 cout<<arr+1; //输出第二个元素的内存地址 以此类推 cout<<*arr+1; //输出11 cout<<*(arr+1); //输出20 指针修改数组元素： 直接通过数组名（指针常量）修改元素 int arr[3] = {10, 20, 30}; *(arr + 1) = 200; // 将第2个元素（下标为1）从20修改为200 通过指针变量修改元素 int arr[3] = {10, 20, 30}; int* p = arr; // 将数组名赋值给指针变量p *(p + 1) = 200; // 通过指针变量p将第2个元素修改为200 两者都能通过 *(指针 + 偏移量) 的方式访问并修改数组中的元素 指针的加减运算是以元素大小为单位移动的，而非以字节为单位。指针 +1 移动到下一个元素，-1 移动到上一个元素 char a[20] = "Hello!"; char *p = a; cout << *p << endl; // 解引用，输出单个字符 → H cout << p << endl; // 输出指针指向的整个字符串 → Hello! p++; // 指针向前移动一个元素，从 a[0] 移到 a[1] cout << *p << endl; // 解引用，输出当前元素 → e cout << p << endl; // 从当前位置输出整个字符串 → ello! int arr[3]={10,20,30}; int *p=NULL; //定义一个空指针 p=arr; // 当前*p=10 p++; //指针向前移动一个字符 cout<<*p<<"\n"; //输出20
